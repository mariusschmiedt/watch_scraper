   function storeSearchStorageModel() {
      if('searchStorageModel' in c24.meta.data) {
         var searchStorage = [];
         var newEntry = c24.meta.data.searchStorageModel;

         try {
            if(localStorage.getItem('searchStorage') != null) {
               searchStorage = JSON.parse(localStorage.getItem('searchStorage'));

               searchStorage.forEach(function(entry, index) {
                  if(newEntry.label === entry.label && newEntry.explanation === entry.explanation) {
                     searchStorage.splice(index, 1);
                  }
               });
            }

            if(String(newEntry.label).trim().length) {
               searchStorage.unshift(newEntry);
            }

            searchStorage = searchStorage.slice(0, 5);

            localStorage.setItem('searchStorage', JSON.stringify(searchStorage));
         } catch (e) {
            // empty
         }
      }
   }

   function getSearchStorage() {
      try {
         return localStorage.getItem('searchStorage') ? JSON.parse(localStorage.getItem('searchStorage')) : [];
      } catch (e) {
         // c24.errors.logWarning('Error in search', 'search-js.jsp.inc', 0, 0, e);
         return [];
      }
   }

   function getSearchStorageAsXml() {
      var xml = "<root>";

      getSearchStorage().forEach(function(entry) {
         if(entry.originalPath) {
            xml += "<option url='" + entry.originalPath + "' filter-text='" + entry.explanation + "'>" + entry.label + "</option>";
         } else {
            xml += "<option url='" + c24.getData('searchIndexUrl') + entry.searchParams + "&dosearch=true' filter-text='" + entry.explanation + "'>" + entry.label + "</option>";
         }
      });

      xml += "</root>";

      return xml;
   }

   function getSelectedTab() {
